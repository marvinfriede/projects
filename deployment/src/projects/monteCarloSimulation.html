def linear(x, y):
    try:
        n = len(x)
        b = (n * sum(x * y) - sum(x) * sum(y)) / (n * sum(x * x) - sum(x)**2)
        a = (sum(y) - b * sum(x)) / n
        return a, b
    except ValueError as err:
        print('ValueError in function "linear()":', err)
        exit(0)


def R2(x, y, a, b, c = 0):
    try:
        n = len(x)
        yMean = sum(y) / n
        yHat = a + b * x + c * x*x

        sst = sum((y - yMean) ** 2)
        sse = sum((yHat - yMean) ** 2)
        #ssr = sum((y - yHat) ** 2)

        r2 = sse / sst
        r2_adj = 1 - (n - 1) / (n - 2) * (1 - r2)

        return r2 * 100 , r2_adj * 100
    except ValueError as err:
        print('ValueError in function "R2()":', err)
        exit(0)
