 29 ''' secant method; x1 and x2 are crucial for finding the desired root '''
 30 def secant(f, x1 = -12345, x2 = 6789, maxiter = 10000, tol = 1e-10):
 31     for iter in range(maxiter):
 32         xnew = x2 - (x2 - x1) / (f(x2) - f(x1)) * f(x2)
 33         if abs(xnew - x2) < tol:
 34             break
 35         else:
 36             x1 = x2
 37             x2 = xnew
 38 
 39     else:
 40         print("Calculation exceeded maximum number of iterations!")
 41         exit()
 42 
 43     return xnew, iter

 46 ''' trapez method for numerical integration; same result for n = 100000 '''
 47 def trapezoidal(f, a, b, n = 10000):
 48     s = 0.0
 49     h = (b - a) / n
 50     for i in  range(0, n):
 51         s += f(a + i * h)
 52 
 53     return h * (s + 0.5 * (f(a) + f(b)))

 56 ''' returns radial SEQ as system of two 1st order differential equations '''
 57 def rad_seq(t, y, energy, veff):
 58     # input: y = [y1, y2]; return y = [y1', y2']
 59     # y1' = y2; y2' = (...)*y1
 60     return [y[1], (- 2 * (energy - veff(t))) * y[0]]

 63 ''' getting initial values for numeric intergration from correct solution '''
 64 def initValues(r):
 65     u = 2 * r * np.exp(-r)
 66     uPrime = (1 - r) * 2 * np.exp(-r)
 67     return [u, uPrime]

 70 ''' wrapper for ODE integration; energy and veff as parameter, integration from
 71 rmax to rmin (inwards) '''
 72 def solve_rad_seq(energy, veff):
 73     sol = solve_ivp(
 74         lambda t, y: rad_seq(t, y, energy, veff),
 75         t_span = [rmax, rmin],
 76         t_eval = np.linspace(rmax, rmin, nsteps),
 77         y0 = initValues(rmax))
 78 
 79     u = sol.y[0]
 80     r = sol.t
 81     return u[::-1], r[::-1]

 84 ''' get first value of integrated Schrödinger equation; since the array is
 85 reversed, u[0] corresponds to the u-value at r = 0 (y-interscetion); different
 86 energies are passed in by secant method '''
 87 def u0(energy, veff):
 88     u, r = solve_rad_seq(energy, veff)
 89     return u[0]

 92 ''' integrating with calculated energy eigenvalue and normalization'''
 93 def normalize(energy, veff):
 94     u, r = solve_rad_seq(energy, veff)
 95     u_spline = Spline(r, u * u)
 96     norm = trapezoidal(u_spline, r[0], r[-1])
 97     u2 = u * u / norm
 98     return u2, r, norm

101 ''' returns poisson equation w''(t) = - u²(t) / t = - density(t) / t as system
102 of two 1st order differential equations '''
103 def poisson(t, y, density):
104     # input: y = [y1, y2]; return y = [y1', y2']
105     # y1' = y2; y2' = - u²(t) / t
106     return [y[1], -density(t) / t]
107 
108 
109 ''' solve poisson equation; input is u²(t) = density(t) from interpolation '''
110 def solve_poisson(density):
111     sol = solve_ivp(
112         lambda t, y: poisson(t, y, density),
113         t_span = [rmin, rmax],
114         t_eval = np.linspace(rmin, rmax, nsteps),
115         y0 = [0, 1])
116 
117     return sol.y[0], sol.t

120 ''' calculate SIC hartree energy '''
121 def sic_hartree_energy(density, vh, r):
122     spline = Spline(r, vh(r) * density(r))
123     ehartree = 0.5 * trapezoidal(spline, r[0], r[-1], maxiter) 
124     return ehartree

127 ''' one iteration in SCF cylce '''
128 def iterStepVerbose(density):
129     # STEP 1: build effective KS potential
130     print("Building KS potential..", end = "", flush = True)
131     step1_start_time = time.time()
132     
133     # solve poisson equation for initial density
134     w, r = solve_poisson(density)
135 
136     # adding homogeneous solution, so that boundary conditions are fulfilled
137     beta = (z - w[-1]) / r[-1]
138     w += beta * r
139 
140     # build SIC Hartree potential (factor 0.5)
141     vh = Spline(r, 0.5 * w / r)
142 
143     step1_exec_time = round((time.time() - step1_start_time), 2)
144     print("\t\t\tbeta = {:.5f}\t\t\t({:.2f}s)".format(beta, step1_exec_time))
145 
146     #########################################################################
147     # STEP 2: solve implicit Schrödinger-like equation
148     print("Solving SP-SEQ..", end = "", flush = True)
149     step2_start_time = time.time()
150 
151     # vext = Spline(r, - z / r ) results in convergence and accuracy problems
152     # vh(r) returns scalar, unlike 0.5 * w / r which returns array -> needed
153     def veff(r):
154         vext = - z / r
155         return vh(r) + vext
156 
157     # find lowest energy eigenvalue; corresponds to energy of hydrogen atom
158     energy, nIter = secant(lambda e: solve_rad_seq(e, veff)[0][0], -1.8, -1.7)
159 
160     step2_exec_time = round((time.time() - step2_start_time), 2)
161     print("\t\t\tE_sp = {: .5f} Eh\t\t({:.2f}s)".format(energy, step2_exec_time))
162 
163 
164     #########################################################################
165     # STEP 3: generate electronic density (and normalization)
166     print("Generating new density..", end = "", flush = True)
167     step3_start_time = time.time()
168 
169     # normalization and generation of electronic density
170     density, r, norm = normalize(energy, veff)
171     new_density = Spline(r, z * density)
172 
173     step3_exec_time = round((time.time() - step3_start_time), 2)
174     print("\t\t\t\t\t\t({:.2f}s)".format(step3_exec_time))
175 
176 
177     #########################################################################
178     # STEP4: accumulate total energy
179     print("Calculating total energy..", end = "", flush = True)
180     step4_start_time = time.time()
181 
182     ehartree = sic_hartree_energy(new_density, vh, r)
183     etot = z * energy - ehartree
184 
185     step4_exec_time = round((time.time() - step4_start_time), 2)
186     print("\t\tE_tot = {: .5f} Eh\t\t({:.2f}s)".format(etot, step4_exec_time))
187 
188     return new_density, etot
189

191 ''' one iteration in SCF cylce, with minimal print output '''
192 def iterStep(density):
193     iter_start_time = time.time()
194     w, r = solve_poisson(density)
195     beta = (z - w[-1]) / r[-1]
196     w += beta * r
197     vh = Spline(r, 0.5 * w / r)
198 
199     def veff(r):
200         vext = - z / r
201         return vh(r) + vext
202 
203     energy, nIter = secant(lambda e: solve_rad_seq(e, veff)[0][0], -1.8, -1.7)
204 
205     density, r, norm = normalize(energy, veff)
206     new_density = Spline(r, z * density)
207 
208     ehartree = sic_hartree_energy(new_density, vh, r)
209     etot = z * energy - ehartree
210 
211     iter_exec_time = round((time.time() - iter_start_time), 2)
212     print("\tE_tot = {: .10f} Eh\t\t({:.2f}s)".format(etot, iter_exec_time))
213 
214     return new_density, etot

217 def main():
218     total_start_time = time.time()
219 
220     # SCF cylce
221     # initializations
222     r = np.linspace(rmin, rmax, nsteps)
223     hydrogen_density = lambda r: z ** 3 / np.pi * np.exp(-2 * z * r)
224     random_density = lambda r: np.random.random() #range(10) # constant function
225 
226     if flags().random:
227         init_density = random_density
228     else:
229         init_density = hydrogen_density   
230 
231     density = old_density = init_density
232     etot_old = 0
233     
234 
235     print("Starting SCF cylce for Helium.")
236     print("------------------------------")
237 
238     if flags().verbose:
239         for iter in range(maxiterSCF):
240             print("Cycle {} of {}".format(iter+1, maxiterSCF))
241             print("--------------")
242             new_density, etot = iterStepVerbose(density)
243 
244             ediff = etot - etot_old
245             print("Change in energy..\t\t\tE_diff = {:.5f} Eh".format(ediff))
246             print("")
247 
248             if abs(ediff) < tol:
249                 total_exec_time = round((time.time() - total_start_time), 2)
250                 print("\nConvergence is achieved! After {} cycles and {:.2f}s.".format(iter+1, total_exec_time))
251                 break
252             else:
253                 # n_mix(r) = a * n_new(r) + (1 - a) * n_old(r)
254                 density_mixed = Spline(r, alpha * new_density(r) + (1 - alpha) * old_density(r))
255 
256                 # density/energy of this step will be old one for next step
257                 old_density = new_density
258                 etot_old = etot
259 
260                 # new density for next iteration
261                 density = density_mixed 
262 
263         else:
264             print("Calculation exceeded maximum number of iterations!")
265             exit()
266 
267     else:
268         for iter in range(maxiterSCF):
269             print("Iteration {}:".format(iter+1), end = "", flush = True)
270             new_density, etot = iterStep(density)
271 
272             if abs(etot - etot_old) < tol:
273                 total_exec_time = round((time.time() - total_start_time), 2)
274                 print("\nConvergence is achieved! After {} cycles and {:.2f}s.".format(iter+1, total_exec_time))
275                 break
276             else:
277                 # n_mix(r) = a * n_new(r) + (1 - a) * n_old(r)
278                 density_mixed = Spline(r, alpha * new_density(r) + (1 - alpha) * old_density(r))
279 
280                 # density/energy of this step will be old one for next step
281                 old_density = new_density
282                 etot_old = etot
283 
284                 # new density for next iteration
285                 density = density_mixed

287 
288         else:
289             print("Calculation exceeded maximum number of iterations!")
290             exit()
