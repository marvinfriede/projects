1 #include <stdio.h>
 2 #include <omp.h>
 3 
 4 int main(void){

23 
24     
25     printf("----------------------------------\n");
26     printf("lastprivate\n\n");

34     }
35

37 
38 
39     printf("----------------------------------\n");
40     printf("firstprivate\n\n");
