def heun3(f, t_eval, y0):
    # stepsize: (tmax - tmin) / #steps
    steps = t_eval.size
    h = abs(t_eval[-1] - t_eval[0]) / steps
    
    # convert to array
    f_ = lambda t, y: np.asarray(f(t, y)) 
    
    # (#steps + 1) rows and (#initial_values) columns; +1 since loop has y[n+1]
    y = np.zeros((steps + 1, len(y0))) 
    
    # first row contains initial values
    y[0] = y0 
    
    # HEUN's third order formula
    for n in range(steps):
        k1 = f_(t_eval[n], y[n])
        k2 = f_(t_eval[n] + 1/3 * h, y[n] + 1/3 * h * k1)
        k3 = f_(t_eval[n] + 2/3 * h, y[n] + 2/3 * h * k2)
        y[n+1] = y[n] + h / 4 * (k1 + 3 * k3)

    return y[:-1,:]


def ode(t, y, k1, k2, k3, k4):
    a, b, c, d, e, f = y    
    a_dot = - k1 * a * b
    b_dot = - k1 * a * b - k2 * b * c
    c_dot = - k3 * c + k4 * f - k2 * b * c
    d_dot = k1 * a * b
    e_dot = k2 * b * c
    f_dot = k3 * c - k4 * f
    return [a_dot, b_dot, c_dot, d_dot, e_dot, f_dot]


def set_k(T, A, Ea):
    # set rate constants according to temperature; a...frequency factors; e...activation energies
    return A * np.exp( -Ea / (8.3145 * T) )


def solver_wrapper(temp, A, Ea, t, y0):
    # solver function: setting k values according to temp and solve with HEUN's 3rd order method
    k1, k2, k3, k4 = set_k(temp, A, Ea)
    return heun3(lambda t, y: ode(t, y, k1, k2, k3, k4), t_eval = t, y0 = y0)

frequency_factors = np.array([6.5031154106E10, 1.5782883877E+11, 1.3305318430E+13, 2.1672258509E+13])
activation_energies = np.array([6.1953762017E04, 8.1032996199E+04, 9.7693597920E+04, 1.0391910534E+05])

# integration range
tmin, tmax = 0, 24
stepsize = tmax * 100
t = np.linspace(tmin, tmax, stepsize)

y0 = [1, 1, 0, 0, 0, 0]
sol = solver_wrapper(298, frequency_factors, activation_energies, t, y0)
a, b, c, d, e, f = sol[:,0], sol[:,1], sol[:,2], sol[:,3], sol[:,4], sol[:,5]

fig = plt.figure() #figsize=(10,10)
plt.xlabel(r"$t$ in h")
plt.ylabel(r"$c$ in mmol l$^{-1}$")
plt.axhline(y=0, color="k", ls="--", lw="0.5")
plt.axhline(y=1, color="k", ls="--", lw="0.5")
plt.axvline(x=0, color="k", ls="--", lw="0.5")
plt.plot(t, b, "r", lw=2, alpha=1, ls="-", label="Valinomycin")
plt.plot(t, a, "g", lw=2, alpha=1, ls="--", label=r"$\mathrm{K}$")
plt.plot(t, d, "#8800cc", lw=2, alpha=1, ls="-", label=r"$[\mathrm{VK}]$")
plt.xlim(tmin - 1, tmax + 1)
plt.legend(loc="best", fancybox=True, shadow=True)
plt.show()
fig.savefig("plotVK.svg", format="svg", transparent=True)
