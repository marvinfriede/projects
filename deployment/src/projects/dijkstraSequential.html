 11 // printing a matrix; for debugging
 12 void printMatrix(int **matrix, int dim){
 13     int i, j;
 14     for (i = 0; i < dim; i++){

 34 void reverseArray(int *arr, int start, int end){ 
 35     int temp; 
 36     while (start < end){ 
 37         temp = arr[start];    
 38         arr[start] = arr[end]; 
 39         arr[end] = temp; 
 40         start++; 
 41         end--; 
 42     }
 43     return;    
 44 }

 47 void printPath(int i, int start, int dimPath, int *path, int *parent, int *distance){
 48     int count, temp, j;
 49 
 50     // removes "d(0-0) = 0    path: 0" from print output
 51     if (i != start){
 52         temp = distance[i];
 53         printf("d(%d-%d) = %d\tpath: ", start, i, temp);
 54 
 55         // catching problem of missing path between nodes, happens for small values of dim
 56         if (temp != INF){
 57             j = i;
 58 
 59             // going backwards through parents until start is reached
 60             count = 0;
 61             while(j != start) {
 62                 j = parent[j];
 63                 path[count] = j;
 64                 count++;
 65             }
 66 
 67             // reversing path for nicer printing (start -> ... -> end)
 68             reverseArray(path, 0, dimPath - 1); 
 69 
 70             // printing the path and resetting path array
 71             for (j = 0; j < dimPath; j++){
 72                 temp = path[j];
 73                 if (temp != INF){
 74                     printf("%d -> ", temp);
 75                 }
 76                 path[j] = INF;
 77             }
 78             printf("%d", i); // end node
 79         }
 80         else {
 81             printf("does not exist.");
 82         }
 83     }
 84     return;
 85 }

 88 // Dijkstra's algorithm; taking a graph, its dimension and the starting point
 89 void dijkstra(int **graph, int dim, int start, clock_t *time_dijkstra){
 90     int i, j, k, minDistance, next = 1, dimPath = 20;
 91     int distance[dim], parent[dim], path[dimPath];
 92     bool visited[dim];
 93 
 94     // loop 1: initializing
 95     for (i = 0; i < dim; i++){
 96         distance[i] = graph[start][i]; // vector of distances from start to adjescent nodes 
 97         parent[i] = 0; // no exisiting parents yet
 98         visited[i] = false; // no nodes visited yet
 99     }
100 
101     // loop 1': initializing path array separately, since dimPath << dim
102     for (i = 0; i < dimPath; i++){
103         path[i] = INF;
104     }
105 
106     distance[start] = 0; // set distance of start node to itself to zero
107     visited[start] = true; // not necessary, since zero is minDistance anyway; saves a bit of time as it circumvents the second if statement in loop 3 
108 
109     // loop 2: iterating over number of nodes
110     for (i = 0; i < dim; i++){
111         minDistance = INF; // initializing minimum distance in every iteration
112 
113         /* loop 3: finding node with lowest distance value in distance vector
114         (if node was not already visited); already processed nodes are
115         excluded according to their visited status (true) */ 
116         for (j = 0; j < dim; j++){
117             if (!visited[j]){
118                 if (distance[j] < minDistance){
119                     minDistance = distance[j]; // minimal distance value
120                     next = j; // index of the closest node
121                 }
122             }
123         }
124         
125         /* closest node in current distance list is found and thus given a
126         processed flag, which will leave it untouched in further iterations*/
127         visited[next] = true;
128 
129         /* loop 4: updating the distance vector with distances from next node 
130         (graph[next][j]) -> including new connections (replacing INF) and
131         shorter indirect paths; by adding the minDistance we keep track of the 
132         distance already travelled w.r.t the start node; already processed
133         nodes must again be excluded */
134         for (j = 0; j < dim; j++){
135             if (!visited[j]){
136                 if (minDistance + graph[next][j] < distance[j]){
137                     distance[j] = minDistance + graph[next][j];
138                     parent[j] = next;
139                 }
140             }
141         }
142     }
143 
144     // time for dijkstra algorithm
145     *time_dijkstra = clock();
146 
147     // printing; only for small values of dim (legibility in terminal, w=160)
148     if (dim < 21){
149         printf("\n");
150         printf("Adjacency matrix:\n-----------------\n");
151         printMatrix(graph, dim);
152         printf("\n\nPaths:\n------");
153         for (i = 0; i < dim; i++){
154             printPath(i, start, dimPath, path, parent, distance);
155             printf("\n");
156         }
157     }
158     // only print path from start to last node
159     else {
160         printf("\nPath (start to last):\n---------------------\n");
161         printPath(dim - 1, start, dimPath, path, parent, distance);
162     }
163     printf("\n");
164 
165     return;
166 }

169 int main(int argc, char *argv[]){
170     int i, j, k, corr = 1, temp, dim;
171     clock_t start_time, end_time, time_init, time_setup, time_dijkstra;
172 
173     // stuff for random numbers; seeding is necessary for true randomness
174     int randMin = 0, randMax;
175     srand(1); // time(NULL)
176 
177     start_time = clock();
178 
179     if (argc != 2){ // really just one argument; default
180         dim = 10000;
181     }
182     else {
183         dim = atoi(argv[1]); // str2int (from stdlib.h)
184     }
185     if (dim > 100){
186         randMax = dim / 10; // dynamic value for longer paths
187     }
188     else {
189         randMax = 10;
190     }
191 
192     // memory allocation for random matrix of dimension <dim>
193     int **randomGraph = (int **)malloc(dim * sizeof(int*));
194     for (i = 0; i < dim; i++){
195         randomGraph[i] = (int *)malloc(dim * sizeof(int));
196     }
197 
198     /* generate matrix of dimension <dim> depicting the adjacency matrix of a
199     (bi)directed graph, which requires a symmetric matrix;
200     initialized with unconnected nodes (INF), i < j gets a triangular matrix
201     without the diagonal (remains INF), upper and lower triangular matrix are
202     filled with the same random values if they are nonzero;
203     faster than a single double loop with if (i < j || i > j) */
204     for (i = 0; i < dim; i++){
205         for (j = 0; j < dim; j++){
206             randomGraph[i][j] = INF;
207         }
208     }
209     time_init = clock();
210     for (i = 0; i < dim; i++){
211         for (j = 0; j < i; j++){ // j < i avoids branch (if statement)
212             temp = randMin + rand() / (RAND_MAX / (randMax - randMin + 1) + 1);
213             if (temp != 0){
214                 randomGraph[i][j] = temp;
215                 randomGraph[j][i] = temp;
216             }
217         }
218     }
219 
220     // time for creating the random graph
221     time_setup = clock();
222 
223     // calling Dijkstra's algorithm
224     dijkstra(randomGraph, dim, 0, &time_dijkstra);
225 
226     // memory deallocation for (2D) arrays
227     for (i = 0; i < dim; i++){
228         free(randomGraph[i]);
229     }
230     free(randomGraph);
231 
232     // total time
233     end_time = clock();
234 
235     // timings
236     printf("\nTimings:\n--------\n");
237     printf("Graph (fill):\t\t%.3lfs\n", (double)(time_init - start_time) / CLOCKS_PER_SEC);
238     printf("Graph (randomize):\t%.3lfs\n", (double)(time_setup - time_init) / CLOCKS_PER_SEC);
239     printf("Dijkstra:\t\t%.3lfs\n", (double)(time_dijkstra - time_setup) / CLOCKS_PER_SEC);
240     printf("Total time:\t\t%.3lfs\n", (double)(end_time - start_time) / CLOCKS_PER_SEC);
241     printf("\n");
242 
243     return 0;
244 }
